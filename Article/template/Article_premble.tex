\usepackage[style=authoryear,autocite=footnote, maxnames=2, doi=false, isbn=false,natbib]{biblatex} 

% To use shortcuts for citing
\newcommand{\ac}{\autocite}
\newcommand{\acv}{\autocite[vgl.]}

% Swap dates from 01.01.2021 to 2021-01-01
\DeclareSourcemap{
    \maps{
        \map[overwrite]{
            \step[fieldsource=urldate,
            match=\regexp{([0-9]{2})\.([0-9]{2})\.([0-9]{4})},
            replace={$3-$2-$1},
            final]
        }
    }
}
